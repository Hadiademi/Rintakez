# Demo Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Run the full app on Vercel with no backend, on realistic mock data, with two fake logins and interactive-in-session writes — behind `NEXT_PUBLIC_DEMO_MODE`, leaving `main`/prod untouched.

**Architecture:** When the flag is on, the four Supabase factories return a mock client backed by an in-memory store seeded from fixtures. The mock client implements the chainable query-builder subset the app uses (incl. FK-hint embedded selects and `count/head`), plus `auth`, `rpc`, and `storage`. All reads/writes go through the server-side store via existing Server Components and Server Actions.

**Tech Stack:** Next.js 16, TypeScript, Vitest. New code under `src/lib/demo/`.

## Task 4: Query builder core (filter/order/insert/update/delete)

**Files:**
- Create: `src/lib/demo/query-builder.ts`
- Test: `src/lib/demo/query-builder.test.ts`

The builder is a thenable object: awaiting it (or calling `.single()/.maybeSingle()`) resolves `{ data, error }`. It accumulates filters then applies them to the store array for the table.

Supported chain: `select(cols, opts?)`, `insert(rows)`, `update(patch)`, `delete()`, `eq`, `neq`, `in`, `is`, `gte`, `lte`, `ilike`, `or`, `contains`, `order`, `range`, `limit`, `single`, `maybeSingle`. Mutations act on the live store array (interactive-in-session).

```ts
import { getStore } from "./store";

type Row = Record<string, any>;
type Filter = (r: Row) => boolean;

export class MockQuery {
  private filters: Filter[] = [];
  private op: "select" | "insert" | "update" | "delete" = "select";
  private payload: any;
  private selectCols = "*";
  private countMode: "exact" | null = null;
  private headOnly = false;
  private orderKey: string | null = null;
  private orderAsc = true;
  private limitN: number | null = null;
  private rangeFromTo: [number, number] | null = null;

  constructor(private table: string) {}

  private get rows(): Row[] {
    const s = getStore() as any;
    if (!s[this.table]) s[this.table] = [];
    return s[this.table];
  }

  select(cols = "*", opts?: { count?: "exact"; head?: boolean }) {
    this.op = this.op === "select" ? "select" : this.op; // insert().select() keeps op
    this.selectCols = cols;
    if (opts?.count) this.countMode = opts.count;
    if (opts?.head) this.headOnly = true;
    return this;
  }
  insert(rows: Row | Row[]) { this.op = "insert"; this.payload = Array.isArray(rows) ? rows : [rows]; return this; }
  update(patch: Row) { this.op = "update"; this.payload = patch; return this; }
  delete() { this.op = "delete"; return this; }

  eq(c: string, v: any) { this.filters.push((r) => r[c] === v); return this; }
  neq(c: string, v: any) { this.filters.push((r) => r[c] !== v); return this; }
  in(c: string, vs: any[]) { this.filters.push((r) => vs.includes(r[c])); return this; }
  is(c: string, v: any) { this.filters.push((r) => r[c] === v); return this; }
  gte(c: string, v: any) { this.filters.push((r) => r[c] >= v); return this; }
  lte(c: string, v: any) { this.filters.push((r) => r[c] <= v); return this; }
  ilike(c: string, p: string) { const re = new RegExp("^" + p.replace(/%/g, ".*") + "$", "i"); this.filters.push((r) => re.test(String(r[c]))); return this; }
  contains(c: string, v: any[]) { this.filters.push((r) => Array.isArray(r[c]) && v.every((x) => r[c].includes(x))); return this; }
  or(expr: string) {
    // supports "col.eq.val,col2.eq.val2"
    const clauses = expr.split(",").map((s) => s.split("."));
    this.filters.push((r) => clauses.some(([c, _op, v]) => String(r[c]) === v));
    return this;
  }
  order(c: string, opts?: { ascending?: boolean }) { this.orderKey = c; this.orderAsc = opts?.ascending ?? true; return this; }
  limit(n: number) { this.limitN = n; return this; }
  range(from: number, to: number) { this.rangeFromTo = [from, to]; return this; }

  private apply(): Row[] {
    let out = this.rows.filter((r) => this.filters.every((f) => f(r)));
    if (this.orderKey) {
      const k = this.orderKey, asc = this.orderAsc;
      out = [...out].sort((a, b) => (a[k] > b[k] ? 1 : a[k] < b[k] ? -1 : 0) * (asc ? 1 : -1));
    }
    if (this.rangeFromTo) out = out.slice(this.rangeFromTo[0], this.rangeFromTo[1] + 1);
    if (this.limitN != null) out = out.slice(0, this.limitN);
    return out;
  }

  private run() {
    // mutations
    if (this.op === "insert") { this.rows.push(...this.payload); return { rows: this.payload }; }
    if (this.op === "update") { const m = this.apply(); m.forEach((r) => Object.assign(r, this.payload)); return { rows: m }; }
    if (this.op === "delete") {
      const m = this.apply(); const live = this.rows;
      for (const r of m) { const i = live.indexOf(r); if (i >= 0) live.splice(i, 1); }
      return { rows: m };
    }
    return { rows: this.apply() };
  }

  single() { const { rows } = this.run(); return Promise.resolve({ data: rows[0] ?? null, error: rows[0] ? null : { message: "no rows" } }); }
  maybeSingle() { const { rows } = this.run(); return Promise.resolve({ data: rows[0] ?? null, error: null }); }
  then(resolve: (v: any) => void) {
    const { rows } = this.run();
    const count = this.countMode ? rows.length : null;
    resolve({ data: this.headOnly ? null : rows, count, error: null });
  }
}

export function from(table: string) { return new MockQuery(table); }
```

## Task 7: Mock RPC

**Files:**
- Create: `src/lib/demo/rpc.ts`
- Test: `src/lib/demo/rpc.test.ts`

Implement against the store, mirroring the SQL functions:
- `accept_bid({ p_bid_id })`: set shoot.status='assigned', shoot.accepted_bid_id=bid, bid.status='accepted', create a conversation (client + bid.photographer) if none; other bids on that shoot → 'declined'.
- `decline_bid({ p_bid_id })`: bid.status='declined'.
- `complete_shoot({ p_shoot_id })`: shoot.status='completed'.
- `get_counterparty_email({ p_shoot_id })`: return the other party's email from personas/profiles.
- `set_initial_role({ p_role })`: set current user's profile.role.
- `shoot_bid_count({ ... })` or returns counts as used by the caller (match its call signature in `src/lib/actions`).

```ts
import { getStore } from "./store";
import { getDemoUserId } from "./session";

export async function rpc(fn: string, args: any = {}) {
  const s = getStore();
  switch (fn) {
    case "accept_bid": {
      const bid = s.bids.find((b: any) => b.id === args.p_bid_id);
      if (!bid) return { data: null, error: { message: "bid not found" } };
      const shoot = s.shoots.find((sh: any) => sh.id === bid.shoot_id);
      shoot.status = "assigned"; shoot.accepted_bid_id = bid.id; bid.status = "accepted";
      s.bids.filter((b: any) => b.shoot_id === shoot.id && b.id !== bid.id).forEach((b: any) => (b.status = "declined"));
      if (!s.conversations.find((c: any) => c.shoot_id === shoot.id))
        s.conversations.push({ id: crypto.randomUUID(), shoot_id: shoot.id, client_id: shoot.client_id, photographer_id: bid.photographer_id, created_at: new Date().toISOString(), last_message_at: new Date().toISOString() });
      return { data: null, error: null };
    }
    // decline_bid, complete_shoot, get_counterparty_email, set_initial_role, shoot_bid_count …
    default: return { data: null, error: null };
  }
}
```
